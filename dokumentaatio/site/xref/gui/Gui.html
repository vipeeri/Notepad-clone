161     public void actionPerformed(ActionEvent e) {
162         JMenuItem choice = (JMenuItem) e.getSource();
163         if (choice == tallennaOsa) {
164             try {
165                 tEditori.tallennaTiedosto(tekstiAlue.getText());
166             } catch (IOException ex) {
167                 Logger.getLogger(Gui.class.getName()).log(Level.SEVERE, null, ex);
168             }
169         }
170         if (choice == tallennaNimella) {
171             try {
172                 tEditori.tallennaNimellaTiedosto(tekstiAlue.getText());
173             } catch (IOException ex) {
174                 Logger.getLogger(Gui.class.getName()).log(Level.SEVERE, null, ex);
175             }
176         }
177         if (choice == avaaOsa) {
178             tEditori.avaa(tekstiAlue);
179         }
180         if (choice == boldaus) {
181             tEditori.lihavoiTeksti(tekstiAlue);
182         }
183         if (choice == arialFontti) {
184             tEditori.vaihdaFontti(tekstiAlue, "arial");
185         }
186         if (choice == serifFontti) {
187             tEditori.vaihdaFontti(tekstiAlue, "serif");
188         }
189         if (choice == vihreaTeema) {
190             tEditori.vaihdaTeema(menuPalkki, "vihrea");
191         }
192     }
