41      public void aloita() throws IOException {
42          Scanner lukija = new Scanner(System.in);
43  
44          while (true) {
45              //Kysytään tiedoston sijainti, ja tekstitiedosto
46              System.out.println("Anna sijainti, esim D:\\: ");
47              this.sijainti = lukija.nextLine();
48              System.out.println("Anna tekstitiedoston nimi: ");
49              this.nimi = lukija.nextLine();
50              //Kysytään mitä tiedostoon kirjoitetaan    
51              System.out.println("Kirjoita tiedostoon: ");
52              this.teksti = lukija.nextLine();
53  
54              if (this.sijainti.equals("") || this.nimi.equals("")) {
55                  System.out.println("Et antanut sijaintia tai nimeä.");
56                  break;
57              } else {
58                  kirjoitaTiedosto();
59              }
60              break;
61          }
62      }

64      public void kirjoitaTiedosto() throws IOException {
65          //Path polku = Paths.get("D:\\", "tässäontekstiä.txt");
66          //String teksti = "\n testikirjoitus";
67          Path polku = Paths.get(this.sijainti, this.nimi);
68          //Tarkistetaan että hakemisto on olemassa
69          Files.createDirectories(polku.getParent());
70          //Kirjoitetaan tiedostoon temp.txt
71          try (
72                  final BufferedWriter kirjoittaja = Files.newBufferedWriter(polku,
73                          StandardCharsets.UTF_8,
74                          StandardOpenOption.CREATE,
75                          StandardOpenOption.APPEND)) {
76              kirjoittaja.write(teksti);
77          } catch (IOException e) {
78              e.printStackTrace();
79  
80          }
81      }
